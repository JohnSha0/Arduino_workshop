#include <Servo.h>
Servo myServo;          // Create servo object
void setup() {
myServo.attach(9);    // Attach servo to pin 9
}
void loop() {
myServo.write(0);     // Move to 0°
delay(2000);
myServo.write(90);    // Move to 90°
delay(2000);
myServo.write(180);   // Move to 180°
delay(2000);
}
